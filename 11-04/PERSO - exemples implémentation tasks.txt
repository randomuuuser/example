### `AgentTask`

```python
# Attributs et méthodes
def web_search(query):
    return f"Results for {query}"

def calculate(expression):
    return eval(expression)

agent_task = AgentTask(
    task_name="research_agent",
    model=gpt4_model,
    tools=[web_search, calculate],  # Attribut
    max_steps=5,                    # Attribut
    memory=conversation_buffer      # Attribut
)

# call_tool()
result = agent_task.call_tool("calculate", {"expression": "2 + 2"})
print(result)  # 4

# run_agent()
final_answer = agent_task.run_agent("Find the capital of France and calculate its population * 2")
print(final_answer)  # "Paris has population 2.1M, doubled is 4.2M"
```

## Audio

### `ASRTask`

```python
# transcribe() avec/sans timestamps
asr_task = ASRTask(
    task_name="transcriber",
    model=whisper_model,
    sample_rate=16000,
    audio_format="wav",
    return_timestamps=True  # Attribut
)

audio = asr_task.load_audio("speech.wav")

# Avec timestamps
segments = asr_task.transcribe(audio)
print(segments[0])  # {"word": "Hello", "start": 0.12, "end": 0.34}

# Sans timestamps
asr_task.return_timestamps = False
text = asr_task.transcribe(audio)
print(text)  # "Hello world how are you today"
```
